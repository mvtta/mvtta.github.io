import matplotlib.pyplot as plt
from scipy.integrate import solve_ivp

# Define oscillator frequencies as powers of phi
n_osc = 5
frequencies = [phi**k for k in range(n_osc)]

# Coupling matrix (nearest neighbor, golden ratio scaling)
K = np.zeros((n_osc, n_osc))
for i in range(n_osc - 1):
    K[i, i+1] = K[i+1, i] = phi

# Kuramoto model dynamics
def kuramoto(t, theta, omega, K):
    dtheta = omega + np.sum(K * np.sin(np.subtract.outer(theta, theta)), axis=1)
    return dtheta

# Initial phases and simulation
theta0 = np.linspace(0, 2*np.pi, n_osc)
t_span = (0, 20)
t_eval = np.linspace(*t_span, 1000)
sol = solve_ivp(kuramoto, t_span, theta0, args=(frequencies, K), t_eval=t_eval)

# Plot results

2 HyperNetX Model for Nautilus Shell Creation

2.1 Hypergraph Representation of Nautilus Growth

Building on our previous analysis of power law scaling and energy-mediated organization, we can model the nautilus shell creation process using HyperNetX to capture the multi-way relationships inherent in biological growth systems. The nautilus shell represents an excellent example of how λ ∝ √DT scaling manifests in natural pattern formation.

2.1.1 Mathematical Foundation

The nautilus shell growth exhibits recursive generative dynamics where each new chamber connects not just to its immediate predecessor, but maintains structural relationships with all previous chambers. This creates a nested hypergraph structure where:

  • Nodes: Individual chambers (C₁, C₂, C₃, …)
  • Hyperedges: Growth events connecting multiple chambers simultaneously
  • Scaling: Each chamber exhibits power law scaling relative to previous chambers

2.1.2 Core Implementation

import hypernetx as hnx
import numpy as np

class NautilusHypergraph:
    def __init__(self, num_chambers=8, growth_ratio=1.618):
        """
        Initialize nautilus hypergraph with golden ratio scaling
        
        Parameters:
        -----------
        num_chambers : int
            Number of chambers to model
        growth_ratio : float
            Growth ratio between successive chambers (default: golden ratio)
        """
        self.num_chambers = num_chambers
        self.phi = growth_ratio  # Golden ratio for natural spirals
        self.chambers = [f'C{i+1}' for i in range(num_chambers)]
        
    def create_growth_hyperedges(self):
        """
        Create hyperedges representing nautilus growth patterns
        Each hyperedge connects a chamber to all structurally relevant predecessors
        """
        hyperedges = {}
        
        # Initial chamber (seed)
        hyperedges['E1_genesis'] = ['C1']
        
        # Growth hyperedges - each chamber connects to geometric predecessors
        for i in range(2, self.num_chambers + 1):
            chamber = f'C{i}'
            
            # Primary growth edge: connects to immediate predecessor
            hyperedges[f'E{i}_primary'] = [f'C{i-1}', chamber]
            
            # Structural support edge: connects to all previous chambers
            # This captures the cumulative structural dependency
            structural_chambers = [f'C{j}' for j in range(1, i)] + [chamber]
            hyperedges[f'E{i}_structural'] = structural_chambers
            
            # Spiral constraint edge: connects to chambers following spiral geometry
            if i > 2:
                # Connect to chambers that maintain spiral integrity
                spiral_predecessors = []
                for j in range(1, i):
                    # Include chambers that contribute to spiral stability
                    if (i - j) <= 3:  # Local spiral constraint
                        spiral_predecessors.append(f'C{j}')
                spiral_predecessors.append(chamber)
                hyperedges[f'E{i}_spiral'] = spiral_predecessors
        
        return hyperedges
    
    def create_scaling_hyperedges(self):
        """
        Create hyperedges representing energy and scaling relationships
        """
        hyperedges = {}
        
        # Energy cascade hyperedges
        for i in range(2, self.num_chambers + 1):
            chamber = f'C{i}'
            
            # Energy levels following power law
            energy_levels = []
            for j in range(max(1, i-2), i+1):  # Include recent chambers
                energy_levels.append(f'C{j}')
            
            if len(energy_levels) > 1:
                hyperedges[f'E{i}_energy'] = energy_levels
        
        return hyperedges
    
    def build_complete_hypergraph(self):
        """
        Build the complete nautilus hypergraph combining all relationship types
        """
        growth_edges = self.create_growth_hyperedges()
        scaling_edges = self.create_scaling_hyperedges()
        
        # Combine all hyperedges
        all_hyperedges = {**growth_edges, **scaling_edges}
        
        # Create HyperNetX hypergraph
        self.hypergraph = hnx.Hypergraph(all_hyperedges)
        
        return self.hypergraph
    
    def analyze_structure(self):
        """
        Analyze the hypergraph structure and scaling properties
        """
        if not hasattr(self, 'hypergraph'):
            self.build_complete_hypergraph()
        
        # Get edge sizes and node degrees using correct HyperNetX API
        edge_sizes = [len(self.hypergraph.edges[edge]) for edge in self.hypergraph.edges]
        node_degrees = [self.hypergraph.degree(node) for node in self.hypergraph.nodes]
        
        analysis = {
            'num_nodes': len(self.hypergraph.nodes),
            'num_edges': len(self.hypergraph.edges),
            'edge_sizes': edge_sizes,
            'node_degrees': node_degrees,
            'max_edge_size': max(edge_sizes) if edge_sizes else 0,
            'nodes': list(self.hypergraph.nodes),
            'edges': list(self.hypergraph.edges)
        };
        
        return analysis

def power_law_chamber_scaling(chamber_index, base_size=1.0, D=1e-9, T=1.0):
    """
    Calculate chamber size using power law scaling
    
    Parameters:
    -----------
    chamber_index : int
        Index of the chamber (1-based)
    base_size : float
        Base chamber size
    D : float
        Diffusion coefficient (energy transfer rate)
    T : float
        Time parameter
    """
    # λ ∝ √DT scaling
    lambda_scale = np.sqrt(D * T * chamber_index)
    
    # Golden ratio geometric scaling
    phi_scale = (1.618 ** (chamber_index - 1))
    
    # Combined scaling with energy conservation
    chamber_size = base_size * lambda_scale * phi_scale
    
    return chamber_size

class ProperFractalNautilus:
    def __init__(self, phi=1.618, D=1e-9, voltage_base=0.1):
        """
        Initialize proper fractal nautilus with spiral geometry
        
        Parameters:
        -----------
        phi : float
            Golden ratio for spiral scaling
        D : float
            Diffusion coefficient for λ ∝ √DT scaling
        voltage_base : float
            Base bioelectric voltage
        """
        self.phi = phi
        self.D = D
        self.voltage_base = voltage_base
        self.golden_angle = 2 * np.pi * (1 - 1/phi)  # ≈ 137.5 degrees
        
    def generate_spiral_chamber(self, n, fractal_level=0, parent_voltage=None):
        """
        Generate a chamber positioned along the logarithmic spiral
        
        Parameters:
        -----------
        n : int
            Chamber number (1-based)
        fractal_level : int
            Current fractal depth
        parent_voltage : float
            Voltage from parent chamber
        """
        if parent_voltage is None:
            parent_voltage = self.voltage_base
        
        # Power law scaling: λ ∝ √DT
        T = n  # Time parameter
        lambda_scale = np.sqrt(self.D * T * n)
        
        # Golden ratio geometric scaling
        phi_scale = self.phi ** (n - 1)
        
        # Combined chamber size
        chamber_size = lambda_scale * phi_scale
        
        # Spiral position calculation
        theta = n * self.golden_angle
        radius = self.phi ** n
        
        # Convert to Cartesian coordinates
        x = radius * np.cos(theta)
        y = radius * np.sin(theta)
        
        # Bioelectric voltage scaling
        voltage = parent_voltage * (1 + 0.15 * n)
        
        # Scale for fractal level
        fractal_scale = (1 / self.phi) ** fractal_level
        
        return {
            'chamber': f'C{n}' if fractal_level == 0 else f'C{n}_F{fractal_level}',
            'size': chamber_size * fractal_scale,
            'voltage': voltage,
            'position': (x * fractal_scale, y * fractal_scale),
            'level': n,
            'fractal_level': fractal_level,
            'theta': theta,
            'radius': radius * fractal_scale
        }
    
    def create_base_spiral(self, num_chambers=8):
        """Create the base nautilus spiral with proper geometry"""
        chambers = []
        hyperedges = {}
        
        # Generate base spiral chambers
        for i in range(1, num_chambers + 1):
            chamber = self.generate_spiral_chamber(i)
            chambers.append(chamber)
            
            # Growth hyperedge: connects to all predecessors
            connected_chambers = [f'C{j}' for j in range(1, i+1)]
            hyperedges[f'E{i}_growth'] = connected_chambers
            
            # Spiral constraint hyperedge: connects to spiral neighbors
            if i > 1:
                spiral_neighbors = []
                for j in range(max(1, i-2), i+1):
                    spiral_neighbors.append(f'C{j}')
                hyperedges[f'E{i}_spiral'] = spiral_neighbors
        
        return chambers, hyperedges
    
    def fractal_expand_spiral(self, base_chambers, base_hyperedges, fractal_level=1):
        """Expand the spiral into fractal copies maintaining geometric scaling"""
        fractal_chambers = base_chambers.copy()
        fractal_hyperedges = base_hyperedges.copy()
        
        # For each chamber, create fractal sub-chambers
        for chamber in base_chambers:
            if chamber['voltage'] > 0.2:  # Only high-voltage chambers generate fractals
                base_level = chamber['level']
                
                # Generate 3 fractal sub-chambers per parent
                for sub_n in range(1, 4):
                    fractal_chamber = self.generate_spiral_chamber(
                        sub_n, 
                        fractal_level=fractal_level, 
                        parent_voltage=chamber['voltage']
                    )
                    
                    # Position relative to parent chamber
                    parent_x, parent_y = chamber['position']
                    fractal_x, fractal_y = fractal_chamber['position']
                    
                    # Offset fractal chambers around parent
                    angle_offset = (sub_n - 1) * 2 * np.pi / 3  # 120° apart
                    offset_radius = chamber['size'] * 0.3
                    
                    final_x = parent_x + fractal_x * 0.3 + offset_radius * np.cos(angle_offset)
                    final_y = parent_y + fractal_y * 0.3 + offset_radius * np.sin(angle_offset)
                    
                    fractal_chamber['position'] = (final_x, final_y)
                    fractal_chamber['parent'] = chamber['chamber']
                    fractal_chamber['chamber'] = f'C{base_level}_{sub_n}_F{fractal_level}'
                    
                    fractal_chambers.append(fractal_chamber)
                    
                    # Create fractal hyperedges
                    parent_name = chamber['chamber']
                    fractal_name = fractal_chamber['chamber']
                    
                    # Connect fractal to parent and siblings
                    fractal_hyperedges[f'F{fractal_level}_{fractal_name}'] = [
                        parent_name, fractal_name
                    ]
        
        return fractal_chambers, fractal_hyperedges
    
    def generate_complete_fractal_nautilus(self, num_base_chambers=8, max_fractal_levels=2):
        """Generate the complete fractal nautilus structure"""
        current_chambers, current_hyperedges = self.create_base_spiral(num_base_chambers)
        
        # Add fractal levels
        for fractal_level in range(1, max_fractal_levels + 1):
            current_chambers, current_hyperedges = self.fractal_expand_spiral(
                current_chambers, current_hyperedges, fractal_level
            )
        
        # Create HyperNetX hypergraph
        hypergraph = hnx.Hypergraph(current_hyperedges)
        
        return current_chambers, current_hyperedges, hypergraph

def analyze_fractal_properties(chambers):
    """Comprehensive analysis of fractal properties"""
    
    # Separate by fractal level
    base_chambers = [c for c in chambers if c['fractal_level'] == 0]
    fractal_chambers = [c for c in chambers if c['fractal_level'] > 0]
    
    # Calculate fractal dimension
    if base_chambers and len(base_chambers) > 2:
        sizes = [c['size'] for c in base_chambers]
        levels = [c['level'] for c in base_chambers]
        
        log_levels = np.log(levels)
        log_sizes = np.log(sizes)
        fractal_dim = np.polyfit(log_levels, log_sizes, 1)[0]
    else:
        fractal_dim = 0
    
    # Voltage analysis
    voltages = [c['voltage'] for c in chambers]
    voltage_range = (min(voltages), max(voltages)) if voltages else (0, 0)
    
    # Spatial analysis
    positions = [c['position'] for c in chambers]
    if positions:
        x_coords = [pos[0] for pos in positions]
        y_coords = [pos[1] for pos in positions]
        spatial_extent = (max(x_coords) - min(x_coords), max(y_coords) - min(y_coords))
    else:
        spatial_extent = (0, 0)
    
    # Growth analysis
    total_chambers = len(chambers)
    fractal_ratio = len(fractal_chambers) / total_chambers if total_chambers > 0 else 0
    
    analysis = {
        'total_chambers': total_chambers,
        'base_chambers': len(base_chambers),
        'fractal_chambers': len(fractal_chambers),
        'fractal_dimension': fractal_dim,
        'voltage_range': voltage_range,
        'spatial_extent': spatial_extent,
        'fractal_ratio': fractal_ratio,
        'max_fractal_level': max([c['fractal_level'] for c in chambers]) if chambers else 0
    }
    
    return analysis
